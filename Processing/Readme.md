```java
void setup(){
  size(600,600);
  dang(6,7,20);
}
void dang(int x, int y, int d){//xyd分别是行数列数和距四边边距
  float rectW = width / x- 2 * d / x; //计算x方向的格子宽
  float rectH = height / y-2 * d / y;//计算y方向的格子高
  for(int i = d; i <=  width - d; i += rectW){
    for(int j = d; j <=  height - d; j += rectH){
      rect(i,j,rectW,rectH);
  }
 }
}
```
